Html Templates in django

In Django, an HTML form is used to collect input from users and send it to the backend via an HTTP request (usually POST or GET). Django processes this data using views and models. Forms can be simple HTML elements or Djangoâ€™s built-in forms.Form or forms.ModelForm.

When to use Html Forms

  • When you need custom styling and full control over form structure.
  • When working with AJAX or JavaScript-based front-end frameworks.

# views.py
from django.shortcuts import render
from .models import Student2

# Function-based view to handle form submission
def index(request):
    # Fetch all stored data to display on the frontend
    students = Student2.objects.all()

    # Check if the request method is POST (form submission)
    if request.method == "POST":
        # Retrieve form data using request.POST
        name = request.POST.get('name')  # Get name from form input
        age = request.POST.get('age')    # Get age from form input
        gender = request.POST.get('gender')  # Get selected gender

        # Retrieve file data using request.FILES
        upload_file = request.FILES.get('upload_file')  # Get uploaded file

        # Store data in the database
        Student2.objects.create(
            name=name,
            age=age,
            gender=gender,
            upload_file=upload_file
        )

        # Debugging: Print received values in the terminal
        print(f"Received: Name={name}, Age={age}, Gender={gender}, File={upload_file}")

    # Render the HTML template with data
    return render(request, 'index.html', {'students': students})

# models.py
from django.db import models

# Model to store student details
class Student2(models.Model):
    # Name field with a max length of 40 characters
    name = models.CharField(max_length=40)

    # Age field (Integer type)
    age = models.IntegerField()

    # Gender field with a max length of 100 characters
    gender = models.CharField(max_length=100)

    # File upload field (stores files in the 'files/' directory inside 'media/')
    upload_file = models.FileField(upload_to="files/", null=True, blank=True)

    # String representation for admin panel
    def __str__(self):
        return self.name
