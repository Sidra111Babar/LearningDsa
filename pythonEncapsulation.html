Encapsulation

Encapsulation is one of the core principles of Object-Oriented Programming (OOP) in Python. It refers to the bundling of data (variables) and methods (functions) that operate on the data into a single unit, typically a class. Encapsulation restricts direct access to some of an object's components, which is a way of preventing unintended interference and misuse.

  • Public Members: Accessible from outside the class.
  • Protected Members: Indicated by a single underscore _. Intended to be used only within the class and its subclasses.
  • Private Members: Indicated by double underscores __. These are not directly accessible outside the class.

class BankAccount:
    def __init__(self, account_holder, balance):
        self.account_holder = account_holder  # Public attribute
        self._balance = balance               # Protected attribute
    
    def deposit(self, amount):
        """Method to deposit money into the account."""
        if amount > 0:
            self._balance += amount
            print(f"Deposited: {amount}")
        else:
            print("Deposit amount must be positive.")
    
    def withdraw(self, amount):
        """Method to withdraw money from the account."""
        if 0 < amount <= self._balance:
            self._balance -= amount
            print(f"Withdrawn: {amount}")
        else:
            print("Insufficient balance or invalid amount.")
    
    def get_balance(self):
        """Public method to get the current balance."""
        return self._balance


# Derived Class
class SavingsAccount(BankAccount):
    def __init__(self, account_holder, balance, interest_rate):
        super().__init__(account_holder, balance)  # Initialize base class attributes
        self.interest_rate = interest_rate         # New attribute specific to SavingsAccount
    
    def add_interest(self):
        """Method to add interest to the balance."""
        interest = self._balance * self.interest_rate / 100
        self._balance += interest
        print(f"Interest of {interest} added. New balance: {self._balance}")


# Creating an instance of SavingsAccount
savings_account = SavingsAccount("Jane Doe", 2000, 5)

# Accessing public methods and protected attribute through subclass
print(f"Account Holder: {savings_account.account_holder}")
savings_account.deposit(500)          # Deposited: 500
savings_account.withdraw(1000)        # Withdrawn: 1000
print(f"Current Balance: {savings_account.get_balance()}")  # Current Balance: 1500

# Adding interest to the balance
savings_account.add_interest()        # Interest of 75.0 added. New balance: 1575.0

# Accessing protected attribute directly within the subclass (allowed)
print(f"Balance (from subclass): {savings_account._balance}")  # 1575.0

# Attempting to access protected attribute directly outside the class
print(savings_account._balance)  # Accessible but not recommended: 1200

class SecureAccount:
    def __init__(self, account_holder, balance):
        self.account_holder = account_holder
        self.__balance = balance  # Private attribute
    
    def deposit(self, amount):
        if amount > 0:
            self.__balance += amount
        else:
            print("Invalid deposit amount.")

    def get_balance(self):
        return self.__balance


# Example
secure_account = SecureAccount("Jane Doe", 2000)
print(secure_account.account_holder)  # Accessible
print(secure_account.get_balance())   # Accessible via public method

# print(secure_account.__balance)  # Raises an AttributeError
# Accessing private attribute (not recommended)
print(secure_account._SecureAccount__balance)  # Accessible using name mangling: 2000
