Using Faker in Django

Faker is a Python library that generates fake data, which is useful for testing, populating databases with sample data, and validating queries or lookups.

from django.db import models

class College(models.Model):
    college_name = models.CharField(max_length=100)
    college_address = models.CharField(max_length=100)

class Student(models.Model):
    gender_choices = (('Male', 'Male'), ('Female', 'Female'))
    
    college = models.ForeignKey(College, on_delete=models.CASCADE, null=True, blank=True)  
    name = models.CharField(max_length=100)
    mobile_number = models.CharField(max_length=12)
    email = models.EmailField(max_length=254)
    gender = models.CharField(max_length=6, choices=gender_choices, default='Male')
    age = models.IntegerField(null=True, blank=True)
    date_of_birth = models.DateField()
    file = models.FileField(upload_to='files/')
    student_bio = models.TextField()
    profile_image = models.ImageField(null=True, blank=True, upload_to='images/')
    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

home/seed_db.py

Inside the home app, create a seed_db.py file to insert fake records.

  • Faker generates random addresses for colleges.
  • A predefined list of college names is used.
  • We cycle through the names using i % len(college_names).

Copied!

Not Copied!


from home.models import College
from faker import Faker

fake = Faker()

def dbSeeder(records=10):
    college_names = ['Superior', 'Punjab', 'UMT', 'Punjab University']
    
    for i in range(records):
        address = fake.address()
        College.objects.create(
            college_name=college_names[i % len(college_names)],  # Cycle through names
            college_address=address
        )

Modify seed_db.py to assign students to colleges:

  1. random.randint(0, colleges.count() - 1) Ensures the index stays within range.
  2. Added a condition to check if colleges exist. Avoids errors when inserting students without existing colleges.
  3. fake.date_of_birth(minimum_age=18, maximum_age=25) Ensures valid date of birth within age range.
  4. fake.phone_number() Generates a realistic phone number.

Not Copied!


import random
from home.models import College, Student
from faker import Faker

fake = Faker('en_IN')  # Faker with Indian locale for better names
def dbSeeder(records=10):
    colleges = College.objects.all()
    
    if not colleges:
        print("No colleges found. Run `dbSeeder()` first to insert colleges.")
        return

    for _ in range(records):
        college_index = random.randint(0, colleges.count() - 1)  # Fix index range
        gender_choice = random.choice(['Male', 'Female'])

        Student.objects.create(
            college=colleges[college_index],  # Assign random college
            name=fake.name(),
            mobile_number=fake.phone_number(),
            email=fake.email(),
            gender=gender_choice,
            age=random.randint(18, 25),
            student_bio=fake.text(),
            date_of_birth=fake.date_of_birth(minimum_age=18, maximum_age=25)  # Ensure valid age
        )
